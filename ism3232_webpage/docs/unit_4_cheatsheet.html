CREATE TABLE IF NOT EXISTS requests ( id INTEGER PRIMARY KEY AUTOINCREMENT, requester TEXT NOT NULL, category TEXT NOT NULL, amount REAL, status TEXT DEFAULT 'Pending', notes TEXT );

INSERT INTO requests (requester, category, amount) VALUES ('Taylor', 'Travel', 1200); SELECT * FROM requests; SELECT * FROM requests WHERE status = 'Pending'; SELECT * FROM requests ORDER BY amount DESC; UPDATE requests SET status = 'Approved' WHERE id = 1; -- WHERE is critical! SELECT status, COUNT(*), SUM(amount) FROM requests GROUP BY status; -- operational report

🐍database.py — 5 REQUIRED FUNCTIONS
import sqlite3 DB_FILE = 'data/requests.db' def create_table(db_file=DB_FILE): with sqlite3.connect(db_file) as conn: conn.execute('''CREATE TABLE IF NOT EXISTS requests ( id INTEGER PRIMARY KEY AUTOINCREMENT, requester TEXT NOT NULL, amount REAL, status TEXT DEFAULT 'Pending')''' ) def add_record(requester, category, amount, notes='', db_file=DB_FILE): with sqlite3.connect(db_file) as conn: conn.execute( 'INSERT INTO requests (requester,category,amount,notes) VALUES (?,?,?,?)', (requester, category, amount, notes) ) def get_all_records(db_file=DB_FILE): with sqlite3.connect(db_file) as conn: conn.row_factory = sqlite3.Row cur = conn.execute( 'SELECT * FROM requests ORDER BY id DESC') return [dict(row) for row in cur.fetchall()] def update_status(record_id, new_status, db_file=DB_FILE): with sqlite3.connect(db_file) as conn: conn.execute( 'UPDATE requests SET status=? WHERE id=?', (new_status, record_id) ) def get_status_report(db_file=DB_FILE): with sqlite3.connect(db_file) as conn: cur = conn.execute( '''SELECT status, COUNT(*) as count, SUM(amount) as total FROM requests GROUP BY status''') return cur.fetchall()
⚠ NEVER use f-strings in SQL. Always use ? placeholders.
✓ Every function takes db_file=DB_FILE — enables tmp_path testing.
🌐STREAMLIT — 5 REQUIRED FEATURES
import streamlit as st from database import create_table, add_record, \ get_all_records, update_status, get_status_report create_table() # idempotent — safe every run st.set_page_config(page_title="My App", layout="wide") st.title("Business Request Tracker") tab1, tab2, tab3, tab4, tab5 = st.tabs( ['Submit', 'View', 'Filter', 'Update', 'Report']) with tab1: # Feature 1 — Add a record requester = st.text_input("Your name") category = st.selectbox("Category", ["Travel", "Equipment", "Software"]) amount = st.number_input("Amount", min_value=0.0) if st.button("Submit"): if requester and amount > 0: add_record(requester, category, amount) st.success(f"Submitted for {requester}") else: st.error("Name and amount required.") with tab2: # Feature 2 — View all records records = get_all_records() if records: st.dataframe(records, use_container_width=True) else: st.info("No records yet.") with tab3: # Feature 3 — Filter f = st.selectbox("Status", ["All","Pending","Approved"]) records = get_all_records() if f != "All": records = [r for r in records if r["status"]==f] st.dataframe(records, use_container_width=True) st.caption(f"{len(records)} record(s) shown") with tab4: # Feature 4 — Update status records = get_all_records() if records: opts = {f"ID {r['id']}: {r['requester']}":r['id'] for r in records} sel = st.selectbox("Record", list(opts.keys())) new = st.selectbox("New status", ["Pending","Approved","Rejected"]) if st.button("Update"): update_status(opts[sel], new) st.success(f"Updated to {new}") with tab5: # Feature 5 — Report report = get_status_report() for row in report: st.metric(row[0], row[1]) # status, count
⚠ Streamlit reruns the entire script on every interaction.
⚠ Build the Update selectbox from get_all_records() — never hardcode IDs.
