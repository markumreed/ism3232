The connection pattern — memorise this

with sqlite3.connect() as conn

The with statement is a context manager. When the block exits — normally or due to an error — the connection is automatically committed and closed. Every database function uses this pattern.

The required pattern and why ? placeholders are mandatory
import sqlite3

DB_FILE = 'data/requests.db'

def some_function(value, db_file=DB_FILE):
    with sqlite3.connect(db_file) as conn:
        conn.execute(
            'SQL WITH ? PLACEHOLDERS',
            (value,)   # values as a tuple
        )

# NEVER do this:
conn.execute(f'INSERT INTO t VALUES ({value})')  # SQL injection risk

# ALWAYS do this:
conn.execute('INSERT INTO t VALUES (?)', (value,))
⚠ f-strings in SQL are dangerous

An f-string in a SQL statement allows any string — including malicious SQL commands — to become part of the query. ? placeholders escape the value automatically. Every SQL statement in this course uses ? — no exceptions.

All five required functions

database.py — type every line
import sqlite3
import os

DB_FILE = 'data/requests.db'

def create_table(db_file=DB_FILE):
    os.makedirs(os.path.dirname(db_file), exist_ok=True)
    with sqlite3.connect(db_file) as conn:
        conn.execute('''
            CREATE TABLE IF NOT EXISTS requests (
                id        INTEGER PRIMARY KEY AUTOINCREMENT,
                requester TEXT    NOT NULL,
                category  TEXT    NOT NULL,
                amount    REAL,
                status    TEXT    DEFAULT 'Pending'
            )
        ''')

def add_record(requester, category, amount, db_file=DB_FILE):
    with sqlite3.connect(db_file) as conn:
        conn.execute(
            'INSERT INTO requests (requester, category, amount) VALUES (?,?,?)',
            (requester, category, amount)
        )

def get_all_records(db_file=DB_FILE):
    with sqlite3.connect(db_file) as conn:
        conn.row_factory = sqlite3.Row
        cur = conn.execute('SELECT * FROM requests ORDER BY id DESC')
        return [dict(row) for row in cur.fetchall()]

def update_status(record_id, new_status, db_file=DB_FILE):
    with sqlite3.connect(db_file) as conn:
        conn.execute('UPDATE requests SET status=? WHERE id=?',
                     (new_status, record_id))

def get_status_report(db_file=DB_FILE):
    with sqlite3.connect(db_file) as conn:
        cur = conn.execute('''
            SELECT status, COUNT(*) AS count, SUM(amount) AS total
            FROM requests GROUP BY status
        ''')
        return cur.fetchall()
row_factory = sqlite3.Row

Without row_factory, sqlite3 returns tuples: row[2] for amount. With row_factory, you get Row objects: row['amount']. dict(row) converts each Row to a plain Python dict — the format Streamlit expects for st.dataframe().

Testing with tmp_path

tests/test_database.py — every test gets a fresh isolated database
from database import create_table, add_record, get_all_records, update_status

def test_add_and_retrieve(tmp_path):
    db = str(tmp_path / 'test.db')   # fresh per test
    create_table(db)
    add_record('Taylor', 'Equipment', 1200.00, db_file=db)
    records = get_all_records(db_file=db)
    assert len(records) == 1
    assert records[0]['requester'] == 'Taylor'

def test_default_status(tmp_path):
    db = str(tmp_path / 'test.db')
    create_table(db)
    add_record('Jordan', 'Travel', 350.00, db_file=db)
    records = get_all_records(db_file=db)
    assert records[0]['status'] == 'Pending'

def test_update_status(tmp_path):
    db = str(tmp_path / 'test.db')
    create_table(db)
    add_record('Morgan', 'Software', 2800.00, db_file=db)
    records = get_all_records(db_file=db)
    update_status(records[0]['id'], 'Approved', db_file=db)
    updated = get_all_records(db_file=db)
    assert updated[0]['status'] == 'Approved'
⚠ Every test must pass db_file=db

If any test uses the default DB_FILE, it writes to your development database, not the isolated test database. The tests then depend on each other's data — fragile and unpredictable.
