sql — five required operations
All five operations
-- CREATE
CREATE TABLE IF NOT EXISTS requests (
    id        INTEGER PRIMARY KEY AUTOINCREMENT,
    requester TEXT    NOT NULL,
    category  TEXT    NOT NULL,
    amount    REAL,
    status    TEXT    DEFAULT 'Pending'
);

-- INSERT
INSERT INTO requests (requester, category, amount)
    VALUES ('Taylor', 'Equipment', 1200.00);

-- SELECT
SELECT * FROM requests WHERE status = 'Pending';
SELECT * FROM requests ORDER BY amount DESC;

-- UPDATE  (WHERE is always required)
UPDATE requests SET status = 'Approved' WHERE id = 1;

-- GROUP BY
SELECT status, COUNT(*) AS n, SUM(amount) AS total
FROM requests GROUP BY status;
✗ UPDATE without WHERE — changes every row
UPDATE requests SET status = 'Approved'
-- EVERY row is now Approved. No undo.
✓ Always include WHERE
UPDATE requests SET status = 'Approved' WHERE id = 1

database.py — the connection pattern
✗ f-strings in SQL — never
conn.execute(f"INSERT INTO t VALUES ('{name}')")
# SQL injection risk — name could be malicious SQL
✓ ? placeholders — always
conn.execute(
    'INSERT INTO requests (requester, amount) VALUES (?, ?)',
    (requester, amount)  # sqlite3 escapes automatically
)
Required function pattern
def get_all_records(db_file=DB_FILE) -> list:
    with sqlite3.connect(db_file) as conn:
        conn.row_factory = sqlite3.Row   # named access
        cur = conn.execute(
            'SELECT * FROM requests ORDER BY id DESC'
        )
        return [dict(row) for row in cur.fetchall()]
row_factory: without it → row[2] (fragile). With it → row['amount'] (readable, robust).

Required AI function pattern
def summarise_request(category: str, description: str) -> str:
    client = anthropic.Anthropic()  # reads ANTHROPIC_API_KEY
    message = client.messages.create(
        model='claude-haiku-4-5-20251001',
        max_tokens=150,
        messages=[{
            'role': 'user',
            'content': (
                f'Summarise this {category} request '
                f'in one sentence:\n\n{description}'
            )  # ← no names, no amounts, no IDs
        }]
    )
    return message.content[0].text.strip()
