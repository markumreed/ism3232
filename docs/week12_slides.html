⌨ models.py — two entity classes + one manager

class Employee:
    def __init__(self, emp_id, name, department, salary):
        self.emp_id     = emp_id
        self.name       = name
        self.department = department
        self.salary     = salary
        self.active     = True

    def give_raise(self, percent):
        self.salary *= (1 + percent / 100)

    def deactivate(self):
        self.active = False

    def is_senior(self):
        return self.salary > 80000

class Department:
    def __init__(self, name, budget):
        self.name       = name
        self.budget     = budget
        self.employees  = []

    def hire(self, employee):
        self.employees.append(employee)

⌨ tests/test_models.py — five required test minimum

from models import Employee, Department

def test_give_raise_increases_salary():
    e = Employee(1, "Alex", "Eng", 70000)
    e.give_raise(10)
    assert e.salary == 77000

def test_deactivate_removes_from_headcount():
    dept = Department("Eng", 300000)
    e = Employee(2, "Sam", "Eng", 60000)
    dept.hire(e)
    e.deactivate()
    assert dept.headcount() == 0

def test_payroll_excludes_inactive():
    dept = Department("Eng", 300000)
    e1 = Employee(3, "Jo", "Eng", 50000)
    e2 = Employee(4, "Pat", "Eng", 55000)
    dept.hire(e1); dept.hire(e2)
    e2.deactivate()
    assert dept.payroll_total() == 50000

def test_over_budget_true():
    dept = Department("Mktg", 100000)
    dept.hire(Employee(5, "Lee", "Mktg", 120000))
    assert dept.over_budget() is True
