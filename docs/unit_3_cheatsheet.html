Complete entity class
APPROVAL_LIMIT = 1000.00

class BusinessRequest:
    """Represents a single purchase request."""

    def __init__(self, req_id: int, requester: str,
                 category: str, amount: float) -> None:
        self.req_id    = req_id       # instance attribute
        self.requester = requester
        self.category  = category
        self.amount    = amount
        self.status    = 'Pending'    # default — not a parameter

    def approve(self) -> None:
        """Mark this request as Approved."""
        self.status = 'Approved'

    def requires_review(self) -> bool:
        """Return True if amount exceeds the limit."""
        return self.amount > APPROVAL_LIMIT

    def __repr__(self) -> str:
        return f'BusinessRequest(id={self.req_id}, status={self.status!r})'
the four OOP mistakes — bad → good
✗ Mistake 1: missing self parameter
def approve():            # TypeError on call
✓ Every method needs self first
def approve(self):        # correct
✗ Mistake 2: local variable instead of self.attribute
def approve(self):
    status = 'Approved'   # local — vanishes immediately
✓ Use self.attribute to persist the change
def approve(self):
    self.status = 'Approved'   # changes the instance
✗ Mistake 3: print() instead of return
def requires_review(self):
    print(self.amount > 1000)  # cannot assert a print
✓ Return the value — now testable
def requires_review(self) -> bool:
    return self.amount > 1000
✗ Mistake 4: class-level list (shared by all instances)
class RequestManager:
    requests = []   # ALL managers share this list!
✓ Instance-level list — inside __init__
class RequestManager:
    def __init__(self):
        self.requests = []   # each manager gets its own
manager class — composition (has-a)
Complete manager class
class RequestManager:
    """Manages a collection of BusinessRequest objects."""

    def __init__(self) -> None:
        self.requests = []   # instance-level!

    def add_request(self, req) -> None:
        self.requests.append(req)

    def list_by_status(self, status: str) -> list:
        return [r for r in self.requests if r.status == status]

    def total_amount(self) -> float:
        return sum(r.amount for r in self.requests)

    def summary_report(self) -> None:
        pending = self.list_by_status('Pending')
        print(f'Total:   {len(self.requests)}')
        print(f'Pending: {len(pending)}')
        print(f'Value:   ${self.total_amount():,.2f}')
inheritance — is-a relationship
Child class with super()
class TravelRequest(BusinessRequest):
    TRAVEL_LIMIT = 500.00

    def __init__(self, req_id, requester, amount, destination):
        super().__init__(req_id, requester, 'Travel', amount)
        self.destination = destination   # new attribute

    def requires_review(self) -> bool:
        return self.amount > self.TRAVEL_LIMIT   # overrides parent
is-a test: "A TravelRequest IS A BusinessRequest." — if that sentence works, inheritance is appropriate. Use has-a (composition) when one class holds another.
