⌨ ai_feature.py — minimal summarisation implementation

import anthropic

def summarise_request(category: str, description: str) -> str:
    """
    Return a one-sentence AI summary of a business request.
    Only category and description are sent — no names or amounts.
    """
    client = anthropic.Anthropic()
    prompt = f"Summarise this {category} business request in one sentence:\n{description}"
    message = client.messages.create(
        model="claude-haiku-4-5-20251001",
        max_tokens=150,
        messages=[{"role": "user", "content": prompt}]
    )
    return message.content[0].text
